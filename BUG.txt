"fun isAuthorizationHeaderValid(authHeader: String?): Boolean {
    if (authHeader == null) {
        return false
    }
    val pattern = Pattern.compile(JWT_REGEX_PATTERN)
    val matcher = pattern.matcher(authHeader)
    return matcher.matches()
